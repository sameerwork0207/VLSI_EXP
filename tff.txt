library IEEE;
use IEEE.STD_LOGIC_1164.ALL;

entity t_flipflop is
    Port (
        clk   : in STD_LOGIC;
        reset : in STD_LOGIC;
        T     : in STD_LOGIC;
        Q     : out STD_LOGIC
    );
end t_flipflop;

architecture Behavioral of t_flipflop is
    signal q_reg : STD_LOGIC := '0';

begin

    process(clk, reset)
    begin
        if reset = '1' then
            q_reg <= '0';

        elsif rising_edge(clk) then
            if T = '1' then
                q_reg <= not q_reg; -- toggle
            else
                q_reg <= q_reg;     -- hold
            end if;
        end if;
    end process;

    Q <= q_reg;

end Behavioral;
